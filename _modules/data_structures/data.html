class Status(Enum):
    NONE = 0
    CREATED = 1
    AFFECTED = 2
    READ = 3
    LESS_THAN = 4  # used for both < and <=
    GREATER_THAN = 5  # used for both > and >=
    EQUAL = 6  # used for ==
    DIFFERENT = 7  # used for !=
    LOOKED_INSIDE = 9  # Used for containers
    REMOVED_INSIDE = 10  # Used to allow for new state update when a deletion occur
